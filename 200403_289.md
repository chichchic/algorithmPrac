Time Complexity: O(n _ m);
Space Complexity: O(n _ m);

class Solution {
public:
    void gameOfLife(vector<vector<int>>& board) {

        int ix[8] = {-1, -1, -1, 0, 0, 1, 1, 1};
        int iy[8] = {-1, 0, 1, -1, 1, -1, 0, 1};

        for(int i = 0; i < board.size(); i++)
        {
            for(int j = 0; j < board[i].size(); j++)
            {
                int naver = 0;
                for(int k = 0; k < 8; k++)
                {
                    int x = i+ix[k];
                    int y = j+iy[k];
                    if(x >= 0 && x < board.size() && y >= 0 && y < board[i].size() && board[x][y] >= 1)
                    {
                        naver++;
                    }
                }
                if(board[i][j])
                {
                    if(naver < 2 || naver > 3)
                        board[i][j] = 2;
                }
                else
                {
                    if(naver == 3)
                        board[i][j] = -1;
                }
            }
        }
        for(int i = 0; i < board.size(); i++)
        {
            for(int j = 0; j < board[i].size(); j++)
            {
                if(board[i][j] == 2)
                    board[i][j] = 0;
                if(board[i][j] == -1)
                    board[i][j] = 1;
            }
        }
    }
};
