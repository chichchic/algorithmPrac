```c++

# Time complexity  : O(n^2)
# Space complexity : O(n^2)

class Solution {
public:
    vector<vector<int>> ans;
    void recur(vector<int>& nums, vector<int> result, vector<bool> chk, const int size, int cnum)
    {
        if(size == cnum) ans.push_back(result);
        for(int i = 0; i < size; i++)
        {
            if(!chk[i])
            {
                chk[i] = true;
                result.push_back(nums[i]);
                recur(nums, result, chk, size, cnum+1);
                chk[i] = false;
                result.pop_back();
            }
        }
    }


    vector<vector<int>> permute(vector<int>& nums) {
        int size = nums.size();
        vector<bool> chk;
        for(int i = 0; i < size; i++)
            chk.push_back(false);
        vector<int> result;
        recur(nums, result, chk, size, 0);
        return ans;
    }
};
```

```c++
class Solution {

    vector<bool> chk;

    void recur(vector<int>& nums, int cnum)

        {
            if(!chk[i])

                result.push_back(nums[i]);
                recur(nums, cnum+1);
                chk[i] = false;
                result.pop_back();
            }
        }
    }


    vector<vector<int>> permute(vector<int>& nums) {
        size = nums.size();
        for(int i = 0; i < size; i++)

        return ans;
    }
};
```

```c++
# Time complexity  : O(n^2)
# Space complexity : O(n)

class Solution {
public:
    vector<vector<int>> permute(vector<int>& nums) {
        vector<vector<int>> ans;
        sort(nums.begin(), nums.end()); // Required this for using next_permutation(or prev_permutation)
        do{
            vector<int> temp;
            for(int i = 0; i < nums.size(); i++)
            {
                temp.push_back(nums[i]);
            }
            ans.push_back(temp);
        }while(next_permutation(nums.begin(), nums.end()));
        return ans;
    }
};
```
